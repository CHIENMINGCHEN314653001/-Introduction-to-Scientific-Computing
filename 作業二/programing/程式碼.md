import numpy as np
from scipy.interpolate import CubicSpline, barycentric_interpolate
from scipy.optimize import minimize_scalar
import matplotlib.pyplot as plt
import time

def runge_function(x):
    """Runge函數: f(x) = 1/(1 + 25x²)"""
    return 1 / (1 + 25 * x**2)

def runge_derivative(x):
    """Runge函數的導數"""
    return -50 * x / (1 + 25 * x**2)**2

def evaluate_error(interpolated_func, true_func, evaluation_points):
    """計算最大絕對誤差"""
    return np.max(np.abs(interpolated_func(evaluation_points) - true_func(evaluation_points)))

def find_min_n_cubic_spline(target_error=1e-10, max_n=1000):
    """使用立方樣條插值找到滿足誤差要求的最小N"""
    print("尋找立方樣條插值的最小N...")

    # 在[-1, 1]上生成密集的評估點
    eval_points = np.linspace(-1, 1, 10000)

    for n in range(5, max_n + 1, 5):  # 每次增加5個節點以加快搜索
        # 生成等距節點
        nodes = np.linspace(-1, 1, n)
        values = runge_function(nodes)

        # 創建立方樣條插值
        cs = CubicSpline(nodes, values)

        # 計算誤差
        error = evaluate_error(cs, runge_function, eval_points)

        if n % 100 == 0:
            print(f"N = {n}, Error = {error:.2e}")

        if error < target_error:
            print(f"找到滿足條件的N: {n}, 誤差: {error:.2e}")
            return n, error

    print(f"在N ≤ {max_n}範圍內未找到滿足條件的解")
    return max_n, error

def find_min_n_chebyshev(target_error=1e-10, max_n=1000):
    """使用Chebyshev節點插值找到滿足誤差要求的最小N"""
    print("尋找Chebyshev節點插值的最小N...")

    # 在[-1, 1]上生成密集的評估點
    eval_points = np.linspace(-1, 1, 10000)

    for n in range(5, max_n + 1, 5):
        # 生成Chebyshev節點
        k = np.arange(1, n + 1)
        chebyshev_nodes = np.cos((2 * k - 1) * np.pi / (2 * n))
        values = runge_function(chebyshev_nodes)

        try:
            # 使用重心插值
            interpolated_values = barycentric_interpolate(chebyshev_nodes, values, eval_points)
            error = np.max(np.abs(interpolated_values - runge_function(eval_points)))

            if n % 100 == 0:
                print(f"N = {n}, Error = {error:.2e}")

            if error < target_error:
                print(f"找到滿足條件的N: {n}, 誤差: {error:.2e}")
                return n, error
        except:
            continue

    print(f"在N ≤ {max_n}範圍內未找到滿足條件的解")
    return max_n, error

def compare_methods():
    """比較兩種方法的性能和效果"""
    target_error = 1e-10

    print("=" * 50)
    print("Runge函數插值分析")
    print("=" * 50)

    # 尋找立方樣條插值的最小N
    start_time = time.time()
    n_cubic, error_cubic = find_min_n_cubic_spline(target_error)
    cubic_time = time.time() - start_time

    print("\n" + "-" * 30)

    # 尋找Chebyshev節點插值的最小N
    start_time = time.time()
    n_chebyshev, error_chebyshev = find_min_n_chebyshev(target_error)
    chebyshev_time = time.time() - start_time

    print("\n" + "=" * 50)
    print("結果比較:")
    print("=" * 50)
    print(f"目標誤差: {target_error:.1e}")
    print(f"立方樣條插值 - 所需N: {n_cubic}, 計算時間: {cubic_time:.3f}秒")
    print(f"Chebyshev節點插值 - 所需N: {n_chebyshev}, 計算時間: {chebyshev_time:.3f}秒")

    # 繪製比較圖
    plot_comparison(n_cubic, n_chebyshev)

def plot_comparison(n_cubic, n_chebyshev):
    """繪製兩種方法的比較圖"""
    x_eval = np.linspace(-1, 1, 1000)
    true_values = runge_function(x_eval)

    plt.figure(figsize=(15, 10))

    # 繪製原始Runge函數
    plt.subplot(2, 2, 1)
    plt.plot(x_eval, true_values, 'k-', linewidth=2, label='Runge Function')
    plt.title('Runge Function: $f(x) = 1/(1 + 25x^2)$')
    plt.xlabel('x')
    plt.ylabel('f(x)')
    plt.grid(True, alpha=0.3)
    plt.legend()

    # 繪製立方樣條插值
    plt.subplot(2, 2, 2)
    nodes_cubic = np.linspace(-1, 1, n_cubic)
    values_cubic = runge_function(nodes_cubic)
    cs = CubicSpline(nodes_cubic, values_cubic)
    cubic_values = cs(x_eval)

    plt.plot(x_eval, true_values, 'k-', linewidth=2, label='True')
    plt.plot(x_eval, cubic_values, 'r--', linewidth=1.5, label='Cubic Spline')
    plt.scatter(nodes_cubic, values_cubic, color='red', s=30, label='Nodes')
    plt.title(f'Cubic Spline Interpolation (N={n_cubic})')
    plt.xlabel('x')
    plt.ylabel('f(x)')
    plt.grid(True, alpha=0.3)
    plt.legend()

    # 繪製Chebyshev節點插值
    plt.subplot(2, 2, 3)
    k = np.arange(1, n_chebyshev + 1)
    chebyshev_nodes = np.cos((2 * k - 1) * np.pi / (2 * n_chebyshev))
    chebyshev_values = runge_function(chebyshev_nodes)
    interpolated_values = barycentric_interpolate(chebyshev_nodes, chebyshev_values, x_eval)

    plt.plot(x_eval, true_values, 'k-', linewidth=2, label='True')
    plt.plot(x_eval, interpolated_values, 'b--', linewidth=1.5, label='Chebyshev')
    plt.scatter(chebyshev_nodes, chebyshev_values, color='blue', s=30, label='Nodes')
    plt.title(f'Chebyshev Nodes Interpolation (N={n_chebyshev})')
    plt.xlabel('x')
    plt.ylabel('f(x)')
    plt.grid(True, alpha=0.3)
    plt.legend()

    # 繪製誤差比較
    plt.subplot(2, 2, 4)
    error_cubic = np.abs(cubic_values - true_values)
    error_chebyshev = np.abs(interpolated_values - true_values)

    plt.semilogy(x_eval, error_cubic, 'r-', label='Cubic Spline Error')
    plt.semilogy(x_eval, error_chebyshev, 'b-', label='Chebyshev Error')
    plt.axhline(y=1e-10, color='gray', linestyle='--', label='Target Error')
    plt.title('Interpolation Errors (Log Scale)')
    plt.xlabel('x')
    plt.ylabel('Absolute Error')
    plt.grid(True, alpha=0.3)
    plt.legend()

    plt.tight_layout()
    plt.savefig('runge_interpolation_comparison.png', dpi=300, bbox_inches='tight')
    plt.show()

if __name__ == "__main__":
    compare_methods()
